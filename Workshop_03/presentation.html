Core Architecture Implementation

🚀 Bot Initialization

class EntertainmentBot: def __init__(self): # Azure OpenAI clients setup self.embedding_client = AzureOpenAI( api_key=self.embedding_api_key, api_version=self.azure_api_version, azure_endpoint=self.azure_api_base ) # ChromaDB vector store self.chroma_client = chromadb.PersistentClient( path="./chroma_db" ) self.collection = self.chroma_client.get_or_create_collection( name="entertainment_knowledge" )

def get_embeddings(self, text): """Generate vector embeddings using Azure OpenAI.""" response = self.embedding_client.embeddings.create( input=text, model=self.embedding_model # text-embedding-3-small ) return response.data[0].embedding

Smart Recommendation Engine

🔍 Semantic Search

def get_recommendations(self, user_input): # Generate query embedding query_embedding = self.get_embeddings(user_input) # Semantic similarity search search_results = self.collection.query( query_embeddings=[query_embedding], n_results=3, include=['documents', 'metadatas'] )

🤖 LLM Generation

# Generate personalized recommendations response = self.llm_client.chat.completions.create( model=self.llm_model, # GPT-4o-mini messages=[ {"role": "system", "content": "You are a friendly entertainment bot"}, {"role": "user", "content": prompt} ], temperature=0.7, max_tokens=500 ) return response.choices[0].message.content

Text-to-Speech Integration

🎵 TTS Model Initialization

def initialize_tts(self): """Initialize HuggingFace SpeechT5 models.""" self.tts_processor = SpeechT5Processor.from_pretrained( "microsoft/speecht5_tts" ) self.tts_model = SpeechT5ForTextToSpeech.from_pretrained( "microsoft/speecht5_tts" ) self.vocoder = SpeechT5HifiGan.from_pretrained( "microsoft/speecht5_hifigan" ) # Load speaker embeddings for voice quality embeddings_dataset = load_dataset( "Matthijs/cmu-arctic-xvectors", split="validation" ) self.speaker_embeddings = torch.tensor( embeddings_dataset[7306]["xvector"] ).unsqueeze(0)

🔊 Generate Audio

def text_to_speech(self, text): """Convert text recommendation to speech.""" # Tokenize and process text inputs = self.tts_processor( text=text, return_tensors="pt" ) # Generate speech with speaker embeddings speech = self.tts_model.generate_speech( inputs["input_ids"], self.speaker_embeddings, vocoder=self.vocoder ) # Save as WAV file with timestamp timestamp = datetime.now().strftime("%Y%m%d_%H%M%S") filename = f"recommendation_{timestamp}.wav" sf.write(filename, speech.numpy(), samplerate=16000) return filename
